üöÄ New Schema Design: restaurant_core

Schema Setup for Supabase

0. Create Schema and Enable Extensions 
-- Create the restaurant_core schema
CREATE SCHEMA IF NOT EXISTS restaurant_core;

-- Enable PostGIS for spatial data (delivery zones)
CREATE EXTENSION IF NOT EXISTS postgis;

-- Enable UUID generation
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- Set search path
SET search_path TO restaurant_core, public;
1. Main Restaurant Entity 
-- Create custom enum types
CREATE TYPE restaurant_status AS ENUM (
    'pending', 
    'active', 
    'suspended', 
    'inactive', 
    'closed'
);

CREATE TABLE restaurant_core.restaurants (
    id SERIAL PRIMARY KEY,
    uuid UUID UNIQUE NOT NULL DEFAULT uuid_generate_v4(),
    slug VARCHAR(255) UNIQUE NOT NULL,
    legacy_v1_id INTEGER,
    legacy_v2_id INTEGER,
    
    -- Basic Information
    name VARCHAR(255) NOT NULL,
    legal_name VARCHAR(255),
    tax_id VARCHAR(50),
    
    -- Ownership
    franchise_id INTEGER,
    parent_restaurant_id INTEGER,
    vendor_id INTEGER,
    
    -- Status
    status restaurant_status NOT NULL DEFAULT 'pending',
    onboarding_completed BOOLEAN DEFAULT FALSE,
    
    -- Timestamps
    created_at TIMESTAMPTZ DEFAULT NOW(),
    created_by INTEGER,
    updated_at TIMESTAMPTZ,
    updated_by INTEGER,
    activated_at TIMESTAMPTZ,
    suspended_at TIMESTAMPTZ,
    closed_at TIMESTAMPTZ
);

-- Create indexes
CREATE INDEX idx_restaurants_status ON restaurant_core.restaurants(status);
CREATE INDEX idx_restaurants_slug ON restaurant_core.restaurants(slug);
CREATE INDEX idx_restaurants_franchise ON restaurant_core.restaurants(franchise_id);
CREATE INDEX idx_restaurants_vendor ON restaurant_core.restaurants(vendor_id);
CREATE INDEX idx_restaurants_legacy ON restaurant_core.restaurants(legacy_v1_id, legacy_v2_id);

-- Add update trigger for updated_at
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$ LANGUAGE plpgsql;

CREATE TRIGGER update_restaurants_updated_at 
BEFORE UPDATE ON restaurant_core.restaurants 
FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
2. Location Management 
CREATE TABLE restaurant_core.restaurant_locations (
    id SERIAL PRIMARY KEY,
    restaurant_id INTEGER NOT NULL REFERENCES restaurant_core.restaurants(id) ON DELETE CASCADE,
    is_primary BOOLEAN DEFAULT FALSE,
    
    -- Address
    street_address VARCHAR(255) NOT NULL,
    unit_number VARCHAR(20),
    city VARCHAR(100) NOT NULL,
    province_id INTEGER,
    postal_code VARCHAR(10),
    country_code CHAR(2) DEFAULT 'CA',
    
    -- Geocoding
    latitude DECIMAL(10, 8),
    longitude DECIMAL(11, 8),
    google_place_id VARCHAR(255),
    
    -- Contact
    phone VARCHAR(20),
    phone_ext VARCHAR(10),
    email VARCHAR(255),
    
    -- Status
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ
);

-- Create indexes
CREATE INDEX idx_locations_restaurant ON restaurant_core.restaurant_locations(restaurant_id);
CREATE INDEX idx_locations_coords ON restaurant_core.restaurant_locations(latitude, longitude);
CREATE INDEX idx_locations_city ON restaurant_core.restaurant_locations(city);

-- Add trigger for updated_at
CREATE TRIGGER update_locations_updated_at 
BEFORE UPDATE ON restaurant_core.restaurant_locations 
FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
3. Service Configuration 
CREATE TABLE restaurant_core.service_configs (
    id SERIAL PRIMARY KEY,
    restaurant_id INTEGER NOT NULL UNIQUE REFERENCES restaurant_core.restaurants(id) ON DELETE CASCADE,
    
    -- Service types
    offers_delivery BOOLEAN DEFAULT FALSE,
    offers_takeout BOOLEAN DEFAULT FALSE,
    offers_dinein BOOLEAN DEFAULT FALSE,
    offers_catering BOOLEAN DEFAULT FALSE,
    
    -- Order settings
    min_order_delivery DECIMAL(8,2),
    min_order_takeout DECIMAL(8,2),
    default_prep_time_delivery INTEGER, -- minutes
    default_prep_time_takeout INTEGER,
    
    -- Advanced settings
    allow_preorders BOOLEAN DEFAULT FALSE,
    preorder_max_days INTEGER DEFAULT 7,
    auto_accept_orders BOOLEAN DEFAULT FALSE,
    require_order_confirmation BOOLEAN DEFAULT TRUE,
    
    -- Capacity management
    max_concurrent_orders INTEGER,
    kitchen_capacity_per_hour INTEGER,
    
    -- Additional settings stored as JSONB for flexibility
    settings JSONB,
    
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ
);

-- Add trigger for updated_at
CREATE TRIGGER update_service_configs_updated_at 
BEFORE UPDATE ON restaurant_core.service_configs 
FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
4. Contact Information 
-- Create contact type enum
CREATE TYPE contact_type AS ENUM (
    'owner', 
    'manager', 
    'chef', 
    'billing', 
    'technical', 
    'marketing'
);

CREATE TABLE restaurant_core.restaurant_contacts (
    id SERIAL PRIMARY KEY,
    restaurant_id INTEGER NOT NULL REFERENCES restaurant_core.restaurants(id) ON DELETE CASCADE,
    
    -- Contact type and person info
    contact_type contact_type NOT NULL,
    first_name VARCHAR(100),
    last_name VARCHAR(100),
    title VARCHAR(100),
    
    -- Contact details
    email VARCHAR(255) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    
    -- Preferences
    receives_orders BOOLEAN DEFAULT FALSE,
    receives_statements BOOLEAN DEFAULT FALSE,
    receives_marketing BOOLEAN DEFAULT FALSE,
    preferred_language CHAR(2) DEFAULT 'en',
    
    -- Status
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ
);

-- Create indexes
CREATE INDEX idx_contacts_restaurant ON restaurant_core.restaurant_contacts(restaurant_id);
CREATE INDEX idx_contacts_email ON restaurant_core.restaurant_contacts(email);
CREATE INDEX idx_contacts_type ON restaurant_core.restaurant_contacts(contact_type);

-- Add trigger for updated_at
CREATE TRIGGER update_contacts_updated_at 
BEFORE UPDATE ON restaurant_core.restaurant_contacts 
FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
5. Delivery Zones with Spatial Support 
CREATE TABLE restaurant_core.delivery_zones (
    id SERIAL PRIMARY KEY,
    restaurant_id INTEGER NOT NULL REFERENCES restaurant_core.restaurants(id) ON DELETE CASCADE,
    zone_name VARCHAR(100),
    zone_number INTEGER,
    
    -- Delivery parameters
    delivery_fee DECIMAL(6,2),
    min_order DECIMAL(6,2),
    estimated_time INTEGER, -- minutes
    
    -- Geographic data with PostGIS
    zone_polygon GEOMETRY(POLYGON, 4326) NOT NULL,
    
    -- Priority (for overlapping zones)
    priority INTEGER DEFAULT 0,
    
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ
);

-- Create spatial index
CREATE INDEX idx_zones_polygon ON restaurant_core.delivery_zones USING GIST (zone_polygon);
CREATE INDEX idx_zones_restaurant ON restaurant_core.delivery_zones(restaurant_id);

-- Add trigger for updated_at
CREATE TRIGGER update_zones_updated_at 
BEFORE UPDATE ON restaurant_core.delivery_zones 
FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
